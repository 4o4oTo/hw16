function [ ] =pl3d( )  % Стартирам функцията pl3d без параметри 
	z = 0:pi/200:2*pi; % Създавам масив от 0 до П*2 със стъпка : П/200 
	x = cos(z);  % Създавам масив със косиносите на стойностите от масива z
	y = sin(z);  % Създавам масив със синосите на стойностите от масива z
 	plot3(x,y,z,'+');  % Създавам триизмерна графика с аргументи масивите x,y,z , която да се 

изписва с '+';
end

>> pl3d()
